    <script>
        let mic;
        let fft;
        let isMicEnabled = false;
        let fireworks = [];
        let particles = [];
        let micButton;
        let volumeBar;
        let frequencyDisplay;
        
        // 音频分析参数
        let bassThreshold = 100;
        let volumeHistory = [];
        const historyLength = 10;
        
        function setup() {
            const canvas = createCanvas(windowWidth * 0.9, 500);
            canvas.parent('p5-canvas');
            
            // 初始化FFT分析器
            fft = new p5.FFT();
            fft.setInput(mic);
            
            // 初始化UI元素
            micButton = select('#mic-button');
            volumeBar = select('#volume-bar');
            frequencyDisplay = select('#frequency-display');
            
            // 设置按钮点击事件
            micButton.mousePressed(toggleMicrophone);
            
            colorMode(HSB, 360, 100, 100, 100);
            noStroke();
        }
        
        function draw() {
            // 创建深色渐变背景
            drawBackground();
            
            if (isMicEnabled && mic) {
                // 分析音频
                let spectrum = fft.analyze();
                let volume = fft.getEnergy(20, 20000) / 255; // 获取整体音量
                let bass = fft.getEnergy(20, 250); // 低频
                let mid = fft.getEnergy(250, 2000); // 中频
                let treble = fft.getEnergy(2000, 20000); // 高频
                
                // 计算平均频率
                let avgFreq = calculateAverageFrequency(spectrum);
                
                // 更新UI显示
                updateAudioDisplay(volume, avgFreq);
                
                // 根据音量决定是否生成新烟花
                if (volume > 0.1) {
                    // 音量越大，烟花越多
                    let fireworkCount = map(volume, 0.1, 1, 1, 5);
                    
                    for (let i = 0; i < fireworkCount; i++) {
                        // 随机延迟生成烟花，使效果更自然
                        if (frameCount % Math.floor(map(volume, 0.1, 1, 30, 5)) === 0) {
                            createFirework(avgFreq, volume, bass, mid, treble);
                        }
                    }
                }
            }
            
            // 更新和显示所有烟花
            updateFireworks();
            updateParticles();
        }
        
        function drawBackground() {
            // 创建星空背景
            for (let i = 0; i < 100; i++) {
                let x = random(width);
                let y = random(height);
                let size = random(0.5, 2);
                fill(255, 255, 255, random(50, 100));
                ellipse(x, y, size, size);
            }
            
            // 添加深色渐变覆盖层
            let gradient = drawingContext.createLinearGradient(0, 0, 0, height);
            gradient.addColorStop(0, 'rgba(10, 10, 40, 0.8)');
            gradient.addColorStop(1, 'rgba(5, 5, 20, 0.9)');
            drawingContext.fillStyle = gradient;
            drawingContext.fillRect(0, 0, width, height);
        }
        
        function calculateAverageFrequency(spectrum) {
            let total = 0;
            let sum = 0;
            
            for (let i = 0; i < spectrum.length; i++) {
                total += spectrum[i];
                sum += spectrum[i] * i;
            }
            
            return total > 0 ? sum / total : 0;
        }
        
        function createFirework(avgFreq, volume, bass, mid, treble) {
            // 根据频率确定烟花颜色
            let hue;
            if (avgFreq < 50) {
                // 低频 - 红色系
                hue = map(avgFreq, 0, 50, 0, 30);
            } else if (avgFreq < 100) {
                // 中频 - 蓝色系
                hue = map(avgFreq, 50, 100, 200, 240);
            } else {
                // 高频 - 黄色系
                hue = map(avgFreq, 100, 255, 50, 70);
            }
            
            // 根据音量确定烟花大小和粒子数量
            let fireworkSize = map(volume, 0.1, 1, 3, 8);
            let particleCount = map(volume, 0.1, 1, 50, 300);
            
            // 在底部随机位置生成烟花
            let x = random(width * 0.1, width * 0.9);
            
            fireworks.push({
                x: x,
                y: height,
                targetY: random(height * 0.2, height * 0.6),
                size: fireworkSize,
                hue: hue,
                saturation: map(mid, 0, 255, 60, 100),
                brightness: map(treble, 0, 255, 80, 100),
                speed: random(5, 10),
                particleCount: particleCount,
                exploded: false,
                particles: []
            });
        }
        
        function updateFireworks() {
            for (let i = fireworks.length - 1; i >= 0; i--) {
                let f = fireworks[i];
                
                if (!f.exploded) {
                    // 烟花上升
                    f.y -= f.speed;
                    
                    // 绘制上升轨迹
                    fill(f.hue, f.saturation, f.brightness, 70);
                    ellipse(f.x, f.y, f.size, f.size);
                    
                    // 添加拖尾效果
                    for (let j = 0; j < 3; j++) {
                        let trailY = f.y + random(5, 15);
                        let trailSize = random(f.size * 0.5, f.size * 0.8);
                        fill(f.hue, f.saturation, f.brightness, 30);
                        ellipse(f.x, trailY, trailSize, trailSize);
                    }
                    
                    // 检查是否到达目标高度或超出屏幕
                    if (f.y <= f.targetY || f.y < 0) {
                        explodeFirework(f);
                        fireworks.splice(i, 1);
                    }
                }
            }
        }
        
        function explodeFirework(firework) {
            // 创建爆炸粒子
            for (let i = 0; i < firework.particleCount; i++) {
                let angle = random(TWO_PI);
                let speed = random(1, 5) * map(firework.particleCount, 50, 300, 0.5, 2);
                
                particles.push({
                    x: firework.x,
                    y: firework.y,
                    vx: cos(angle) * speed,
                    vy: sin(angle) * speed,
                    life: 100,
                    maxLife: 100,
                    hue: firework.hue + random(-20, 20),
                    saturation: firework.saturation,
                    brightness: firework.brightness,
                    size: random(2, 6)
                });
            }
        }
        
        function updateParticles() {
            for (let i = particles.length - 1; i >= 0; i--) {
                let p = particles[i];
                
                // 更新位置
                p.x += p.vx;
                p.y += p.vy;
                
                // 添加重力效果
                p.vy += 0.05;
                
                // 减少生命值
                p.life -= 1;
                
                // 计算透明度和大小
                let alpha = map(p.life, 0, p.maxLife, 0, 100);
                let currentSize = map(p.life, 0, p.maxLife, 0, p.size);
                
                // 绘制粒子
                fill(p.hue % 360, p.saturation, p.brightness, alpha);
                ellipse(p.x, p.y, currentSize, currentSize);
                
                // 移除死亡粒子
                if (p.life <= 0 || p.y > height || p.x < 0 || p.x > width) {
                    particles.splice(i, 1);
                }
            }
        }
        
        function updateAudioDisplay(volume, avgFreq) {
            // 更新音量条
            let volumePercent = volume * 100;
            volumeBar.style('width', volumePercent + '%');
            
            // 更新频率显示
            let freqValue = map(avgFreq, 0, 255, 0, 20000);
            frequencyDisplay.html(`频率: ${Math.round(freqValue)} Hz`);
            
            // 根据音量改变音量条颜色
            if (volume < 0.3) {
                volumeBar.style('background', 'linear-gradient(90deg, #33ccff, #33ccff)');
            } else if (volume < 0.6) {
                volumeBar.style('background', 'linear-gradient(90deg, #33ccff, #ff9933)');
            } else {
                volumeBar.style('background', 'linear-gradient(90deg, #33ccff, #ff3366)');
            }
        }
        
        function toggleMicrophone() {
            if (!isMicEnabled) {
                // 启用麦克风
                mic = new p5.AudioIn();
                mic.start(() => {
                    isMicEnabled = true;
                    micButton.html('麦克风已启用');
                    micButton.style('background', 'linear-gradient(90deg, #33cc66, #33ccff)');
                    
                    // 将麦克风输入连接到FFT
                    fft.setInput(mic);
                }, (err) => {
                    console.error('麦克风访问错误:', err);
                    alert('无法访问麦克风。请确保已授予麦克风权限。');
                });
            } else {
                // 禁用麦克风
                if (mic) {
                    mic.stop();
                }
                isMicEnabled = false;
                micButton.html('启用麦克风');
                micButton.style('background', 'linear-gradient(90deg, #ff3366, #3366ff)');
                
                // 清空音量显示
                volumeBar.style('width', '0%');
                frequencyDisplay.html('频率: -- Hz');
            }
        }
        
        function windowResized() {
            resizeCanvas(windowWidth * 0.9, 500);
        }
    </script>
